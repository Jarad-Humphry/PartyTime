	protected int findIndex(String word) {
		

		int i;
		int index = hashFunction(word);
		int k = 1;
		incProbeCount();

		if (table[index] == null)
			{
			return index;
			}
		
		if (table[index].getWord().equals(word))
			{
			return index;
			}

		i = index;
		i++;
		if (i >= table.length)
			{
			i = 0;
			}
		
		while (table[i]!=null)

			{
			incProbeCount();

			if (table[i].getWord().equals(word))
				{
				return i;
				}
			
			if (i==index)
				{
				return -1;
				}
			else
				{
				i ++;
				if (i >= table.length)
					{
					i = 0;
					}
				}
			}
		return i;
		
		
	}
